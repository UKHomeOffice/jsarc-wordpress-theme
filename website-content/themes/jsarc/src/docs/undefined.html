scss 
// $color__jsarc-primary-blue: #1d4189;
// $color__jsarc-primary-red: #d8262e;
// $color__jsarc-teal: #0594a6;
// $color__jsarc-light-blue: #06a6e0;
// $color__jsarc-purple: #972d45;
// $color__jsarc-light-green: #95ccaa;
// $colors: (
//   // set explicit colors
//   "jsarc-primary-blue": #1d4189,
//   "jsarc-primary-red": #d8262e,
//   "jsarc-teal": #0594a6,
//   "jsarc-light-blue":  #06a6e0,
//   "jsarc-purple": #972d45,
//   "jsarc-light-green": #95ccaa,
// );
/// @colors
$brand-colors: (
  "jsarc-primary-blue": #1d4189,
  "jsarc-primary-red": #d8262e,
  "jsarc-teal": #0594a6,
  "jsarc-light-blue":  #06a6e0,
  "jsarc-purple": #972d45,
  "jsarc-light-green": #95ccaa,
);
$ui-colors: (
  "background": "#fff",
  "border": "#333",
  "link": "#jsarc-light-blue"
);
$typographic-colors: (
  "bodyText": "#brand-light",
  "caption": "#brand-dark",
  "secondaryText": "#brand-pink" ("shade": 25%),
);

// merge everything into the main $colors map…
///colors
@include add-colors($brand-colors, $ui-colors, $typographic-colors);
$colors: (
  // set explicit colors
  "brand-pink": hsl(330, 85%, 62%),
  "brand-light": #ddf,
  "brand-dark": #224,

  // reference existing colors
  "background": "#brand-light",
  "border": "#brand-dark",

  // make adjustments as needed, using color functions
  "link": "#brand-pink" ("shade": 25%, "desaturate": 15%),
);

.example {
  border-color: color("border");
}
